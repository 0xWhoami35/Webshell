        <form method="post">
            <input type="text" name="directory" placeholder="Path Directory" required>
            <input type="text" name="filename" placeholder="Name of File to Edit" required>
            <textarea name="newtext" placeholder="Text to Prepend to File" required></textarea>
            <button type="submit">Submit</button>
        </form>

        <?php
        if ($_SERVER['REQUEST_METHOD'] == 'POST') {
            $directory = rtrim($_POST['directory'], '/');
            $filename = $_POST['filename'];
            $newtext = $_POST['newtext'];
            $filepath = $directory . '/' . $filename;

            if (file_exists($filepath)) {
                $fileContent = file_get_contents($filepath);
                $updatedContent = $newtext . $fileContent;
                file_put_contents($filepath, $updatedContent);
                echo "<div class='output'>Prepended text to file: $filepath</div>";
            } else {
                echo "<div class='output'>File does not exist: $filepath</div>";
            }
        }
        ?>
